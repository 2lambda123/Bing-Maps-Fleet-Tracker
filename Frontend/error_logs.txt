/**
 * Writes the error logs to the "error_logs.txt" file.
 * @param errorLogs - The error logs to be written.
 */
function writeErrorLogsToFile(errorLogs: string): void {
  writeErrorLogsToFile(errorLogs);
}

const errorLogs = `Error 1: Something went wrong.
Error 2: Another error occurred.`;
